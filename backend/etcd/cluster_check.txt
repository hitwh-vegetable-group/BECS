source /opt/k8s/bin/environment.sh

  do
    echo ">>> ${node_ip}"
    ETCDCTL_API=3 /opt/k8s/bin/etcdctl \
    --endpoints=https://${node_ip}:2379 \
    --cacert=/etc/kubernetes/cert/ca.pem \

    --key=/etc/etcd/cert/etcd-key.pem endpoint health
  done
